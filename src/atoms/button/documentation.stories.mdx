### Theming

<Canvas>
  <Story name='Themed button'>
    {() => {
      const theme = buildTheme({
        buttons: {
          default: {
            normal: {
              default: {
                background: {
                  'background-color': '$colors.brandSecondaryLight50',
                },
                text: {
                  color: '$colors.brandPrimary'
                }
              },
              hover: {
                background: {

              },
              press: {
