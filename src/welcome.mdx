## Quick Start

```bash
npm install @kibalabs/ui-react
```

```tsx
// Import styles
import '@kibalabs/ui-react/dist/index.css';

// Use components
import { Button, Stack, Text } from '@kibalabs/ui-react';

function App() {
  return (
    <Stack direction="vertical" gap="md">
      <Text variant="header1">Hello World</Text>
      <Button variant="primary" text="Get Started" />
    </Stack>
  );
}
```

## Customize

Override CSS variables to match your brand:

```scss
:root {
  --kiba-color-brand-primary: #your-brand-color;
  --kiba-font-family: 'Your Font', sans-serif;
  --kiba-border-radius: 8px;
}
```
