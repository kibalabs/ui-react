# TabBar

<br/>

export const Template = (args) => {
  const [selectedTabKey, setSelectedTabKey] = React.useState();
  return (
    <TabBar {...args} selectedTabKey={args.selectedTabKey || selectedTabKey} onTabKeySelected={args.setSelectedTabKey || setSelectedTabKey}>
      <TabBar.Item tabKey={String(1)} text='First' />
      <TabBar.Item tabKey={String(2)} text='Second' />
      <TabBar.Item tabKey={String(3)} text='Third' />
    </TabBar>
  )
};

<Canvas withToolbar={true} withSource={'none'}>
  <Story name='Default' args={{}}>
    {Template.bind({})}
  </Story>
</Canvas>
