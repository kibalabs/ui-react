# OptionSelect

<br/>

export const Template = (args) => {
  const [selectedItemKey, setSelectedItemKey] = React.useState('');
  const options=[{itemKey: '1' , text: 'First Option'},{itemKey: '2' , text: 'Second Option'},{itemKey: '3' , text: 'Third Option'}];
  return <OptionSelect {...args} selectedItemKey={args.selectedItemKey || selectedItemKey} options={args.options || options} onItemClicked={args.onItemClicked || setSelectedItemKey} />;
};

<Canvas withToolbar={true} withSource={'none'}>
  <Story name='Default' args={{}}>
    {Template.bind({})}
  </Story>
</Canvas>
