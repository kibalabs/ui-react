# List

<br/>

export const Template = (args) => {
  const [selectedItemKey, setSelectedItemKey] = React.useState();
  return (
    <List {...args} selectedItemKey={args.selectedItemKey || selectedItemKey} onItemClicked={args.setSelectedItemKey || setSelectedItemKey}>
      <List.Item itemKey={String(1)}>First</List.Item>
      <List.Item itemKey={String(2)}>Second</List.Item>
      <List.Item itemKey={String(3)}>Third</List.Item>
    </List>
  )
};

<Canvas withToolbar={true} withSource={'none'}>
  <Story name='Default' args={{}}>
    {Template.bind({})}
  </Story>
</Canvas>
