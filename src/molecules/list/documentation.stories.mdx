export const Template = (args) => {
  const [selectedListItemKey, setSelectedListItemKey] = React.useState();
  return (
    <List {...args} selectedListItemKey={args.selectedListItemKey || selectedListItemKey} onListItemKeySelected={args.setSelectedListItemKey || setSelectedListItemKey}>
      <List.Item listItemKey={String(1)}>First</List.Item>
      <List.Item listItemKey={String(2)}>Second</List.Item>
      <List.Item listItemKey={String(3)}>Third</List.Item>
    </List>
  )
};
