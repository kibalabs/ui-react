# Examples
<Canvas >
  <Story name='Message Dialog'>
    {() => {
      const [openDialog, setOpenDialog] = React.useState(false);
      return (
        <React.Fragment>
          <Button variant='primary' onClicked={() => setOpenDialog(true)} text='Open Dialog' />
          <MessageDialog 
            isOpen={openDialog}
            title='Message Dialog'
            message='This is a simple message dialog with two  buttons'
            onConfirmClicked={() => {console.log('confirm close'); setOpenDialog(!openDialog)}}
            onCloseClicked={() => setOpenDialog(!openDialog)}
          />
        </React.Fragment>
      );
    }}
  </Story>
